\section{DFT不同算法运行时间对比}

\subsection{按照 DFT 的定义，使用 for 循环计算}
\begin{lstlisting}[language=matlab]
for k = 0:N-1
	for n = 0:N-1
		X(k+1) = X(k+1) + x(n+1) * WN ^ (k * n);
	end
end
\end{lstlisting}

\subsection{使用 DFT 的矩阵形式计算}
\begin{lstlisting}[language=matlab]
P = exp(-1i * 2 * pi / length(x) * (0:length(x)-1).' * (0:length(x)-1));
X = x * P;
\end{lstlisting}



\subsection{调用 MATLAB 的 fft 函数计算}
\begin{lstlisting}[language=matlab]
X = fft(x);
\end{lstlisting}

\subsection{调用 MATLAB 的 fft 函数，并且将序列搬移到 GPU 上计算}
\begin{lstlisting}[language=matlab]
x_GPU = gpuArray(x);
X = fft(x_GPU);
\end{lstlisting}

\subsection{分析DFT不同算法运行时间对比}
\begin{itemize}
	\item 按照定义直接计算DFT，需要嵌套的两次循环，时间复杂度为$O(N^2)$；
	\item 使用矩阵形式计算DFT，时间复杂度为$O(N^2)$。但是MATLAB对矩阵乘法有所优化，所以计算速度比直接计算DFT快；
	\item 调用MATLAB的fft函数计算DFT，时间复杂度为$O(N\log N)$，速度比直接计算和矩阵计算快；
	\item 调用MATLAB的fft函数在GPU上计算DFT，由于数据在GPU上并行处理，所以速度比CPU上的fft函数快。
\end{itemize}
